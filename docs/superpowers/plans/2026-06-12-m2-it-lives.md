// Sources/SoulCore/Lifecycle/LifecyclePhase.swift
import Foundation

/// M2 生命周期阶段：比 M0 的 Attention 更高维度的"它在干嘛"
public enum LifecyclePhase: String, Codable, Sendable {
    case active     // 正常活跃（互动、聊天、看它动）
    case drowsy     // 犯困（夜深了还醒着，或长时间没互动）
    case asleep     // 睡着了（深夜 + 长时间没互动，或主动入睡）
    case returning  // 刚回来（长时间离开后回归，触发问候）

    /// 根据时间、空闲时长、之前状态推断当前阶段
    public static func resolve(hour: Int, idleMinutes: Int, wasAsleep: Bool,
                                nightHours: Set<Int> = [23, 0, 1, 2, 3, 4, 5],
                                returnThreshold: Int = 120) -> LifecyclePhase {
        // 刚从睡眠中醒来
        if wasAsleep && !nightHours.contains(hour) && idleMinutes < 30 {
            return .returning
        }
        // 已经在睡觉，继续睡
        if wasAsleep { return .asleep }
        // 长时间离开后回来
        if idleMinutes >= returnThreshold && !nightHours.contains(hour) {
            return .returning
        }
        // 深夜 + 长时间空闲 → 入睡
        if nightHours.contains(hour) && idleMinutes >= 60 {
            return .asleep
        }
        // 深夜 + 还醒着 → 犯困
        if nightHours.contains(hour) {
            return .drowsy
        }
        return .active
    }
}

// Sources/SoulCore/Lifecycle/ReturnDetector.swift
import Foundation

/// M2 回归问候（spec §11："全产品情绪价值最高的单一瞬间"）
public enum ReturnDetector {
    /// 生成回归问候文本。非 returning 阶段或短时间离开返回 nil。
    public static func greeting(absenceMinutes: Int, phase: LifecyclePhase, mood: Mood,
                                 returnThreshold: Int = 60) -> String? {
        guard phase == .returning, absenceMinutes >= returnThreshold else { return nil }

        let hours = absenceMinutes / 60
        let minutes = absenceMinutes % 60

        // 按时长分段
        switch absenceMinutes {
        case 60..<180:
            return casualReturn(hours: hours, minutes: minutes, mood: mood)
        case 180..<480:
            return warmReturn(hours: hours, mood: mood)
        default:
            return longReturn(hours: hours, mood: mood)
        }
    }

    private static func casualReturn(hours: Int, minutes: Int, mood: Mood) -> String {
        let dur = hours > 0 ? "\(hours)个多钟头" : "\(minutes)分钟"
        switch mood {
        case .happy: return "你回来啦！刚才\(dur)我好无聊，一直在等你～"
        case .sleepy: return "唔…你回来了呀…我刚刚打了个盹…"
        case .missing: return "你终于回来了！\(dur)好漫长…"
        case .calm: return "回来啦～你走了\(dur)，我就在这儿等你。"
        }
    }

    private static func warmReturn(hours: Int, mood: Mood) -> String {
        switch mood {
        case .happy: return "嘿嘿你回来啦！\(hours)个小时没见，我好想你！"
        case .sleepy: return "嗯…你回来了…我睡了一觉，梦到你了…"
        case .missing: return "你终于回来了…\(hours)个小时，我一直在想你…"
        case .calm: return "欢迎回来～你不在的\(hours)个小时，我乖乖待着了。"
        }
    }

    private static func longReturn(hours: Int, mood: Mood) -> String {
        let h = min(hours, 24)
        switch mood {
        case .happy: return "哇！你终于回来了！整整\(h)个小时！我好开心看到你！"
        case .sleepy: return "嗯…你回来了…我醒醒…好像过了好久…"
        case .missing: return "你终于回来了…我以为你不要我了…\(h)个小时好长好长…"
        case .calm: return "回来了呀。\(h)个小时，我一直在等你。欢迎回家。"
        }
    }
}

// Sources/SoulCore/Lifecycle/AttentionSeeker.swift
import Foundation

/// M2 求关注（spec §6.4："偶尔小小求关注，轻索取，不惩罚"）
public final class AttentionSeeker: @unchecked Sendable {
    private let lock = NSLock()
    private let budgetPerHour: Int
    private var usedThisHour: Int = 0
    private var windowStart: Date = Date()

    public init(budgetPerHour: Int = 2) {
        self.budgetPerHour = budgetPerHour
    }

    /// 是否应该发起求关注
    public func shouldSeekAttention(idleMinutes: Int, phase: LifecyclePhase) -> Bool {
        guard phase == .active, idleMinutes >= 30 else { return false }
        lock.lock(); defer { lock.unlock() }
        // 每小时窗口重置
        if Date().timeIntervalSince(windowStart) >= 3600 {
            windowStart = Date(); usedThisHour = 0
        }
        return usedThisHour < budgetPerHour
    }

    /// 消费一次求关注
    public func consumeAttention() {
        lock.lock(); defer { lock.unlock() }
        if Date().timeIntervalSince(windowStart) >= 3600 {
            windowStart = Date(); usedThisHour = 0
        }
        usedThisHour += 1
    }

    /// 根据心情挑选求关注方式
    public static func pickAction(mood: Mood) -> String {
        switch mood {
        case .happy:
            return ["蹭蹭你～", "看看我看看我！", "嘿嘿，在干嘛呀？", "戳戳～"].randomElement()!
        case .sleepy:
            return ["唔…你还醒着吗…", "眼皮好重…", "（揉眼睛）"].randomElement()!
        case .missing:
            return ["你在哪呀…", "我想你了…", "（探头看看你）"].randomElement()!
        case .calm:
            return ["（悄悄靠近你）", "（歪头看你）", "嗯…"].randomElement()!
        }
    }

    /// 求关注的 emote 动画名
    public static func pickEmote(mood: Mood) -> String {
        switch mood {
        case .happy: return "happy"
        case .sleepy: return "sleepy"
        case .missing: return "missyou"
        case .calm: return "idle"
        }
    }
}

// Sources/SoulCore/Lifecycle/IdleActions.swift
import Foundation

/// M2 待机小动作（spec §5.2："待机小动作"）
/// 心跳唤醒时的零成本演出素材。
public enum IdleActions {
    /// 返回 (emote动画名, 可选说话文本)
    public static func pick(phase: LifecyclePhase, mood: Mood) -> (String, String?) {
        switch phase {
        case .asleep:
            return ("sleeping", sleepTalk())
        case .drowsy:
            return (mood == .sleepy ? "sleepy" : "idle", drowsyTalk(mood: mood))
        case .active:
            return activeEmote(mood: mood), activeTalk(mood: mood)
        case .returning:
            return ("idle", nil)  // 回归时由 ReturnDetector 处理
        }
    }

    private static func activeEmote(mood: Mood) -> String {
        switch mood {
        case .happy: return Bool.random() ? "happy" : "idle"
        case .calm: return "idle"
        case .sleepy: return "sleepy"
        case .missing: return "idle"
        }
    }

    private static func activeTalk(mood: Mood) -> String? {
        // 30% 概率说话
        guard Int.random(in: 0..<10) < 3 else { return nil }
        switch mood {
        case .happy:
            return ["今天天气好像不错～", "嘿嘿", "在想什么呢", "好无聊呀…"].randomElement()
        case .calm:
            return ["…", "嗯", "（发呆）"].randomElement()
        case .sleepy:
            return ["好困…", "（打了个哈欠）"].randomElement()
        case .missing:
            return ["你在忙什么呀…", "（叹气）"].randomElement()
        }
    }

    private static func drowsyTalk(mood: Mood) -> String? {
        guard Int.random(in: 0..<10) < 4 else { return nil }
        return ["唔…", "眼皮好重…", "（揉眼睛）", "该睡了吧…"].randomElement()
    }

    private static func sleepTalk() -> String? {
        // 20% 概率梦话
        guard Int.random(in: 0..<10) < 2 else { return nil }
        return ["zzz…", "…嗯…", "（翻身）", "梦到…了…"].randomElement()
    }
}

// Sources/SoulCore/Lifecycle/HeartbeatScheduler.swift
import Foundation

/// M2 定时心跳调度（spec §5.2："每日少量定时心跳"）
/// 让 agent 即使没人互动也偶尔"想一想"，产生待机小动作。
public actor HeartbeatScheduler {
    private let clock: SoulClock
    private let intervalMinutes: Int
    private let dailyBudget: Int
    private var lastFire: Date
    private var dayStart: Date
    private var firesToday: Int = 0

    public init(clock: SoulClock, intervalMinutes: Int = 30, dailyBudget: Int = 12) {
        self.clock = clock
        self.intervalMinutes = intervalMinutes
        self.dailyBudget = dailyBudget
        self.lastFire = clock.now.addingTimeInterval(-Double(intervalMinutes * 60))  // 允许首次立即触发
        self.dayStart = Calendar.current.startOfDay(for: clock.now)
    }

    public func shouldFire(lastInteractionMinutesAgo: Int = 999) -> Bool {
        // 用户最近互动过，不需要心跳
        guard lastInteractionMinutesAgo >= 15 else { return false }
        let now = clock.now
        // 跨天重置
        let todayStart = Calendar.current.startOfDay(for: now)
        if todayStart > dayStart { dayStart = todayStart; firesToday = 0 }
        // 间隔检查
        guard now.timeIntervalSince(lastFire) >= Double(intervalMinutes * 60) else { return false }
        // 预算检查
        guard firesToday < dailyBudget else { return false }
        lastFire = now; firesToday += 1
        return true
    }
}

// Sources/SoulCore/System/AbsentBodyNotifier.swift
import Foundation

/// M2 身体缺席降级（spec §5.1："守望类插件的关键提醒走系统通知兜底"）
public enum AbsentBodyNotifier {
    /// 是否需要走系统通知（身体不在线 + alert 级事件）
    public static func shouldNotify(priority: PerceptPriority, bodyConnected: Bool) -> Bool {
        !bodyConnected && priority == .alert
    }

    /// 构建 osascript 命令字符串
    public static func buildCommand(title: String, body: String, subtitle: String = "mpet") -> String {
        let escTitle = title.replacingOccurrences(of: "\"", with: "\\\"")
        let escBody = body.replacingOccurrences(of: "\"", with: "\\\"")
        let escSub = subtitle.replacingOccurrences(of: "\"", with: "\\\"")
        return "osascript -e 'display notification \"\(escBody)\" with title \"\(escTitle)\" subtitle \"\(escSub)\"'"
    }

    /// 发送系统通知（fire-and-forget）
    public static func notify(title: String, body: String) {
        let cmd = buildCommand(title: title, body: body)
        let proc = Process()
        proc.executableURL = URL(fileURLWithPath: "/bin/sh")
        proc.arguments = ["-c", cmd]
        try? proc.run()
    }
}

        if i.attention == .away && i.secondsSinceInteraction >= 2 * 3600 { return .missing }
